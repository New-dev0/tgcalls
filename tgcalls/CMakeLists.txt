    ${src_loc}/WrappedAudioDeviceModuleImpl.h

    ${src_loc}/FileAudioDevice.cpp

    ${src_loc}/NativeInstance.h

  target_compile_features(tgcalls
  PUBLIC
      cxx_std_20

    lib_tgcalls
    external_ffmpeg
